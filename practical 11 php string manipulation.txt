        echo "Reverse: " . strrev($str) . "<br>";

        echo "Uppercase: " . strtoupper($str) . "<br>";

        echo "Lowercase: " . strtolower($str) . "<br>";

        echo "Substring (first 5 chars): "
             . substr($str, 0, 5) . "<br>";

        echo "Word Count: "
             . str_word_count($str) . "<br>";

        echo "Replace 'a' with '@': "
             . str_replace('a', '@', $str) . "<br>";
